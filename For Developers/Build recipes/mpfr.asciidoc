#!/bin/bash

# recipe for:
P=mpfr
V=3.1.5

# download
curl -L \
	--output "${DOWNLOADS_DIR}/$P-$V.tar.gz" \
	"http://www.mpfr.org/$P-$V/$P-$V.tar.gz"

# unpack
mkdir -p "${SRC_DIR}/$P" && cd "$_"
tar xvf "${DOWNLOADS_DIR}/$P-$V.tar.gz"

# configure
mkdir -p "${SRC_DIR}/$P/build" && cd "$_"
"${SRC_DIR}/$P/$P-$V/configure" \
		--prefix="${PREFIX}" \
		--with-gmp="${PREFIX}" \
		--disable-shared

# compile & install
make -j4
make install
