#!/bin/bash
#
# **Set package and version**
#
P=cmake
V=3.10.0

#
# **Download CMake**
#
curl -L \
         --output "${DOWNLOADS_DIR}/$P-$V.tar.gz" \
        "https://cmake.org/files/v${V%.*}/$P-$V.tar.gz"
#
# **Unpack CMake**
#
mkdir -p "${SRC_DIR}/$P" && cd "$_"
tar xvf "${DOWNLOADS_DIR}/$P-$V.tar.gz"

#
# **Configure CMake**
# You might have to add the option --system-curl
#
mkdir -p "${SRC_DIR}/$P/build" && cd "$_"
"${SRC_DIR}/cmake/cmake-$V/bootstrap" \
        --prefix="${PREFIX}" \
        --verbose
#
# **Compile and install CMake**
#
make -j ${NJOBS}
make install
